pragma solidity ^0.8.16;

/* this will allow one, in remix, to get the interface IDs of the various interfaces, which is needed for the grading.js grading script for this assignment */

import "./IERC165.sol";
import "./IERC721full.sol";
import "./IERC20Metadata.sol";

contract InterfaceIDs {
    bytes4 public idIERC20;
    bytes4 public idIERC20Metadata;
    bytes4 public idIERC165;
    bytes4 public idIERC721Metadata;
    bytes4 public idIERC721full;
    bytes4 public idIERC721;

    constructor() {
        // for TokenCC
        idIERC165 = type(IERC165).interfaceId;
        idIERC20 = type(IERC20).interfaceId;
        idIERC20Metadata = type(IERC20Metadata).interfaceId;
        // for NFTmanager (IERC165 is repeated intentionally)
        idIERC165 = type(IERC165).interfaceId;
        idIERC721 = type(IERC721).interfaceId;
        idIERC721Metadata = type(IERC721Metadata).interfaceId;
        idIERC721full = type(IERC721full).interfaceId;
    }
}
